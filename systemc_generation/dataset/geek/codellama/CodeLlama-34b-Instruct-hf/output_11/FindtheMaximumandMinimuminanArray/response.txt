SC_MODULE(ArrayMinMax) {
    sc_in<int> input_array[]; // Input array
    sc_in<int> array_size; // Size of the input array
    sc_out<int> min_val; // Output for the minimum value
    sc_out<int> max_val; // Output for the maximum value

    // Constructor
    SC_CTOR(ArrayMinMax) {
        // Process to find the minimum and maximum elements
        SC_METHOD(find_min_max);
        sensitive << input_array << array_size;

    // Method to find the minimum and maximum elements in the array
    void find_min_max() {
        int min_index = 0, max_index = 0;
        for (int i = 0; i < array_size; i++) {
            if (input_array[i] < input_array[min_index]) {
                min_index = i;
            }
            if (input_array[i] > input_array[max_index]) {
                max_index = i;
            }
        }
        min_val.write(input_array[min_index]);
