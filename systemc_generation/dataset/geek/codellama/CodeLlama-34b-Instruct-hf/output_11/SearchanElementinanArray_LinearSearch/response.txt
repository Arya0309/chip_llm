// Module that implements linear search
SC_MODULE(LinearSearch) {
    sc_in<int> key;  // Input port for the key
    sc_in<int> data;  // Input port for the array
    sc_out<int> index;  // Output port for the found index

    // Constructor
    SC_CTOR(LinearSearch) {
        // Process to perform linear search
        SC_METHOD(search);
        sensitive << key << data;
    }

    // Method to perform linear search
    void search() {
        int i = 0;
        while (i < data.read().size()) {
            if (data.read()[i] == key.read()) {
                index.write(i);
                break;
            }
            i++;
        }
        if (i == data.read().size()) {
            index.write(-1);
        }
    }
};
