    sc_in<int> arr;
    sc_in<int> n;

    void findMinMax() {
        int min = INT_MAX;
        int max = INT_MIN;
        for (int i = 0; i < n; i++) {
            if (arr[i] < min) {
                min = arr[i];
            }
            if (arr[i] > max) {
                max = arr[i];
            }
        }
        min_val.write(min);
        max_val.write(max);
    }
};
