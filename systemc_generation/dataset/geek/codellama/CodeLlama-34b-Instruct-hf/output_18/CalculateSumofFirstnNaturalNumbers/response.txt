SC_MODULE(Summation) {
    sc_in<int> n;  // Input port for the number of natural numbers to sum
    sc_out<int> result;  // Output port for the sum

    // Constructor
    SC_CTOR(Summation) {

    // Method to find the sum of the first n natural numbers
    void findSum() {
        int sum = 0;
        for (int i = 1; i <= n; i++) {
            sum += i;
        }
        result.write(sum);
    }
