    sc_in<int> c;
    sc_out<int> largest;

        int input_b = (int) b.read();
        int input_c = (int) c.read();

        if (input_a >= input_b) {
            if (input_a >= input_c) {

            } else {
                largest.write(input_c);
            }
        } else {
            if (input_b >= input_c) {
                largest.write(input_b);
            } else {
                largest.write(input_c);
            }
        }
    }
};
