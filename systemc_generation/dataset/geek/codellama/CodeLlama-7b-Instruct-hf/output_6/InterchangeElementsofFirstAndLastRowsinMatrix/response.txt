This program takes a square matrix (n x n) as input, where n is a constant defined as 4. The program then interchanges the elements of the first and last column of the matrix. Finally, it outputs the resulting matrix to the console.

#include <systemc.h>

// Module for swapping the elements of the first and 
// last column of a square matrix (n x n)
SC_MODULE(ColumnSwap) {
    sc_in<int> in[n][n]; // Input port for the matrix
    sc_out<int> out[n][n]; // Output port for the swapped matrix

    // Function to swap the elements of the first and 
    // last column of the matrix
    void interchange(int m[][n], int n) {
        // Swapping the elements of the first and last 
        // columns
        for (int i = 0; i < n; i++) {
            int t = m[i][0];
            m[i][0] = m[i][n - 1];
            m[i][n - 1] = t;
        }
    }

    // Driver function
    void main() {
        // Creating a sample array
        int m[n][n] = {{8, 9, 7, 6},
                     {4, 7, 6, 5},
                     {3, 2, 1, 8},
                     {9, 9, 7, 7}};

        interchange(m, n);

        // Printing the swapped matrix
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                out[i][j].write(m[i][j]);
            }
        }
    }
};
